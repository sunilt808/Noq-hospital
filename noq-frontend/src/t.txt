    <script>
        // Print Token Function
        function printToken() {
            const token = document.querySelector('.token-certificate');
            const originalContent = document.body.innerHTML;
            
            document.body.innerHTML = token.outerHTML;
            window.print();
            document.body.innerHTML = originalContent;
            location.reload();
        }
        
        // Emergency Call Function
        function emergencyCall() {
            if (confirm('🚨 Are you in a critical emergency? Click OK to call 108')) {
                window.location.href = 'tel:108';
            }
        }
        
        // Dynamic Date/Time Update
        function updateDateTime() {
            const now = new Date();
            const dateStr = now.toLocaleDateString('en-US', { 
                weekday: 'long', 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            });
            const timeStr = now.toLocaleTimeString('en-US', { 
                hour: '2-digit', 
                minute: '2-digit' 
            });
            
            // Update token generation time
            document.querySelectorAll('.token-row')[5].querySelector('.token-value').textContent = timeStr;
            document.querySelectorAll('.token-row')[6].querySelector('.token-value').textContent = dateStr;
        }
        
        // Update time every minute
        updateDateTime();
        setInterval(updateDateTime, 60000);
        
        // Add hover effects
        document.querySelectorAll('.token-row').forEach(row => {
            row.addEventListener('mouseenter', function() {
                this.style.transform = 'translateY(-3px)';
            });
            
            row.addEventListener('mouseleave', function() {
                this.style.transform = 'translateY(0)';
            });
        });
        
        // Add click to copy token number
        document.querySelectorAll('.token-row')[1].addEventListener('click', function() {
            const tokenNum = this.querySelector('.token-value').textContent;
            navigator.clipboard.writeText(tokenNum).then(() => {
                alert(`Token number ${tokenNum} copied to clipboard!`);
            });
        });
    </script>
